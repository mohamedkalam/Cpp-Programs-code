#include<iostream>
using namespace std;
int main(){
    int a[10][10],b[10][10],multi[10][10],r,c,i,j,k;
    cout<<"Enter the Rows:";
    cin>>r;
    cout<<"Enter the Columns:";
    cin>>c;
    cout<<"First matrix"<<"\n";
    for(i=0;i<c;i++){
      for(j=0;j<r;j++){
        cin>>a[i][j];
      }
    }
    cout<<"Second matrix"<<"\n";
     for(i=0;i<c;i++){
      for(j=0;j<r;j++){
        cin>>b[i][j];
      }
    }
    cout<<"Multi matrix"<<"\n";
     for(i=0;i<c;i++){
      for(j=0;j<r;j++){
        multi[i][j]=0;
           for(k=0;k<c;k++){
             multi[i][j]+=a[i][k]*b[k][j];
           }
      }
    }
     for(i=0;i<c;i++){
      for(j=0;j<r;j++){
          cout<<multi[i][j]<<" ";
      }
      cout<<"\n";
    }

}
